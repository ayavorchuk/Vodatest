        function handleClick(e) {
            const baseGain = gameState.perClick * gameState.dropMultiplier;
            const totalGain = Math.floor(baseGain);
            
            gameState.score += totalGain;
            gameState.totalClicks++;
            gameState.stats.totalEarned += totalGain;
            
            updateDisplay();
            updateCPS();
            checkAchievements();
            checkDropUnlocks();
            createClickEffect(e, totalGain);
            saveGame();
        }

        function selectDropType(type) {
            if (!dropTypes[type]) return;
            
            gameState.currentDropType = type;
            gameState.dropMultiplier = dropTypes[type].multiplier;
            
            // Обновляем визуал капли

            });
            document.querySelector(`[data-type="${type}"]`).classList.add('active');
            
            updateDisplay();
        }
        
        function checkDropUnlocks() {
            Object.entries(dropTypes).forEach(([type, data]) => {
                if (!gameState.unlockedDrops.includes(type) && 
                    gameState.stats.totalEarned >= data.unlockRequirement) {
                    
                    gameState.unlockedDrops.push(type);
                    unlockDrop(type);
                }
            });
        }
        
        function unlockDrop(type) {
            const dropElement = document.querySelector(`[data-type="${type}"]`);
            dropElement.classList.remove('drop-type-locked');
            dropElement.dataset.unlocked = "true";
            
            // Показываем уведомление
            alert(`🎉 Разблокирована новая капля: ${dropTypes[type].name}!`);
        }

        function checkAchievements() {
            achievements.forEach(ach => {
                if (gameState.achievements[ach.id].unlocked) return;
                
                let currentValue = 0;
                switch(ach.type) {
                    case 'clicks':
                        currentValue = gameState.totalClicks;
                        break;
                    case 'score':
                        currentValue = gameState.stats.totalEarned;
                        break;
                    case 'cps':
                        currentValue = gameState.stats.maxCPS;
                        break;
                    case 'upgrades':
                        currentValue = gameState.level - 1; // начальный уровень 1
                        break;
                }
                
                // Обновляем прогресс
                const progress = Math.min(currentValue / ach.requirement * 100, 100);
                gameState.achievements[ach.id].progress = progress;
                
                // Проверяем разблокировку
                if (currentValue >= ach.requirement) {
                    unlockAchievement(ach.id);
                }
            });
            
            renderAchievements();
        }
        
        function unlockAchievement(achievementId) {
            const ach = achievements.find(a => a.id === achievementId);
            if (!ach || gameState.achievements[achievementId].unlocked) return;
            
            gameState.achievements[achievementId].unlocked = true;
            gameState.score += ach.reward;
            
            // Показываем уведомление
            alert(`🏆 Достижение разблокировано!\n\n${ach.icon} ${ach.name}\n${ach.description}\n\nНаграда: +${ach.reward} капель`);
        }
        
        function renderAchievements() {
            achievementsPanel.innerHTML = '';
            
            achievements.forEach(ach => {
                const progress = gameState.achievements[ach.id].progress;
                const unlocked = gameState.achievements[ach.id].unlocked;
                
                const achElement = document.createElement('div');
                achElement.className = `achievement ${unlocked ? 'unlocked' : ''}`;
                achElement.innerHTML = `
                    <div class="achievement-icon">${ach.icon}</div>
                    <div class="achievement-info">
                        <h4>${ach.name}</h4>
                        <p>${ach.description}</p>
                        <div class="achievement-progress">
                            <div class="achievement-progress-bar" style="width: ${progress}%"></div>
                        </div>
                    </div>
                `;
                
                achievementsPanel.appendChild(achElement);
            });
        }

        function upgradeClick() {
            const cost = 50 * Math.pow(1.5, gameState.perClick - 1);
            
            if (gameState.score >= cost) {
                gameState.score -= cost;
                gameState.perClick++;
                gameState.level++;
                updateDisplay();
                saveGame();
            }
        }
        
        function upgradeAutoClicker() {
            const cost = 200 * Math.pow(2, gameState.autoClickers);
            
            if (gameState.score >= cost) {
                gameState.score -= cost;
                gameState.autoClickers++;
                updateDisplay();
                startAutoClickers();
                saveGame();
            }
        }
        
        function upgradeDropValue() {
            const cost = 500 * Math.pow(1.8, Math.floor(gameState.dropMultiplier * 2));
            
            if (gameState.score >= cost) {
                gameState.score -= cost;
                gameState.dropMultiplier += 0.5;
                updateDisplay();
                saveGame();
            }
        }

                    // Создаем эффект автоклика

        function updateCPS() {
            const now = Date.now();
            lastClicks.push(now);
            
            // Удаляем старые клики
            lastClicks = lastClicks.filter(time => now - time < 1000);
            
            const cps = lastClicks.length;
            cpsElement.textContent = cps;
            
            // Обновляем максимальный CPS
            if (cps > gameState.stats.maxCPS) {
                gameState.stats.maxCPS = cps;
            }
        }

        function createClickEffect(e, gain) {

            flyNum.style.position = 'fixed';

            flyNum.style.top = `${e.clientY}px`;
            flyNum.style.color = '#6df2ff';
            flyNum.style.fontWeight = 'bold';
            flyNum.style.fontSize = '1.8rem';
            flyNum.style.textShadow = '0 0 8px rgba(109, 242, 255, 0.9)';
            flyNum.style.pointerEvents = 'none';
            flyNum.style.zIndex = '10000';
            document.body.appendChild(flyNum);
            
            // Анимация полёта
            const flyX = (Math.random() - 0.5) * 120;
            flyNum.animate([
                { transform: 'translate(0, 0) scale(1)', opacity: 1 },
                { transform: `translate(${flyX}px, -80px) scale(0.7)`, opacity: 0 }
            ], {
                duration: 1000,
                easing: 'cubic-bezier(0.2, 0.8, 0.8, 1)'
            });
            
            // Удаление через 1 секунду
            setTimeout(() => flyNum.remove(), 1000);
        }
